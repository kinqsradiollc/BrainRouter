```typescript
function executeWithBackoff<T>(fn: () => T, retries = 3): T {
  try {
    return fn();
  } catch (err) {
    if (retries === 0) throw err;
    const delay = Math.pow(2, 4 - retries) * 100;
    // sleep(delay)
    return executeWithBackoff(fn, retries - 1);
  }
}
```

Applied this wrapper around all `memoryEngine` write transactions in `engine.ts` and set `PRAGMA journal_mode=WAL` during db init.
